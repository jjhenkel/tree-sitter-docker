===================================
Permissive JSON array handling 1
===================================

# Support extra char at end
CMD ["We see this"];
CMD [ 'and unfortunately,', 'this'  ]%

---

(dockerfile
  (comment)
  (directive (cmd 
    (json_array (json_array_item) (json_array_extraneous_char))
  ))
  (directive (cmd 
    (json_array (json_array_item) (json_array_item) (json_array_extraneous_char))
  ))
)
